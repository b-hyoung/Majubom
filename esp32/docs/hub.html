전달 박형석 → 김도경 (서버) 명세

CSI 측에서 60초마다 POST로 보낼 JSON 형식. 자세한 내용 → 시스템_데이터흐름_웹명세.md | CSI_JSON_명세_상세.md (한글 해석 + 시나리오) 

POST /csi   // 침대 환자 상태를 60초마다 서버에 전송
{
  "timestamp": "2026-06-15T14:30:00Z",   // 측정 시점 (UTC 국제표준 시간)
  "bed_id":    "bed_01",                  // 침대 번호 (환자 식별자)
  "sensor":    "csi",                     // 센서 종류 ("csi" 고정)

  "raw": {                                  // ─── 실제 측정값 ───
    "hr_bpm": 73,                          // 심박수 (1분당 박동) — 정상 60~80
    "resp_rpm": 14,                        // 호흡수 (1분당 호흡) — 정상 12~20
    "autocorr_strength": 0.45             // 신호 품질 (0~1) — 높을수록 깨끗
  },

  "zscore": {                              // ─── 평소 대비 변화량 ───
    "hr": 0.4,                              // HR 변화 (+ 빠름 / − 느림)
    "resp": -0.3,                           // 호흡 변화
    "strength": 0.2,                        // 신호 안정성 변화
    "total_abs": 0.9                        // |z| 합계 = 종합 위험 점수
  },

  "presence": {                            // ─── 방에 몇 명? (작업 2) ───
    "count": 1,                              // 인원수 (1 = 환자만 / 2 = 다른 사람도)
    "confidence": 0.96                      // AI 분류 확률 (96% 확신)
  },

  "alert_level": "normal"                // 🟢 normal / 🟡 caution / 🟠 warning / 🔴 critical
}

2~4
